------------- 통신사회원가입 : 핸드폰 인증해야하니깐 핸드폰은 꼭 입력됨. 마이디바이스 등록 3개까지. 1=일반회원,0=탈퇴회원,9=관리자, 2=실버,3=골드,4=VIP 1년,6개월등 실적에따라 변경.----------------
CREATE TABLE MEMBER(
	M_SEQ NUMBER PRIMARY KEY,
	M_ID VARCHAR2(50) NOT NULL UNIQUE,
	M_PW VARCHAR2(50) NOT NULL,
	M_NAME VARCHAR2(50) NOT NULL,
	M_PHONE1 VARCHAR2(3) NOT NULL,
	M_PHONE2 VARCHAR2(4) NOT NULL,
	M_PHONE3 VARCHAR2(4) NOT NULL,
	M_BIRTH VARCHAR2(20) NOT NULL,
	M_ADDRESS1 VARCHAR2(200) NOT NULL,
	M_ADDRESS2 VARCHAR2(200),
	M_MYDEVICE1 VARCHAR2(10),
	M_MYDEVICE2 VARCHAR2(10),
	M_MYDEVICE3 VARCHAR2(10),
	M_STATUS NUMBER DEFAULT(1),
	M_REGDATE DATE NOT NULL
);
CREATE SEQUENCE MEMBER_SEQ;
------------- 관리자기기등록 : NAME은 기기명(갤럭시), MODLENUM은 모델명(MS23JD), REGDATE가 널이면 출시예정표시 ----------------
CREATE TABLE DEVICE(
	D_SEQ NUMBER PRIMARY KEY,
	D_NAME VARCHAR2(50) NOT NULL,
	D_MODELNUM VARCHAR2(10) NOT NULL UNIQUE,
	D_REGDATE DATE,
	D_TYPE VARCHAR2(30) NOT NULL
);
CREATE SEQUENCE DEVICE_SEQ;
---------- 기기 정보 : COLOR별로 상품등록. PRICE가 널인 경우 화면에서 출시예정,금액추후예정 등 표시-----------------
CREATE TABLE DEVICEINFO(
	DI_SEQ NUMBER PRIMARY KEY,
	D_SEQ NUMBER NOT NULL,
	DI_COLOR VARCHAR2(20) NOT NULL,
	DI_IMAGE BLOB,
	DI_IMAGENAME VARCHAR2(200),
	DI_QUANTITY NUMBER DEFAULT(0),
	DI_PRICE NUMBER,
	CONSTRAINT DEVICEINFO_FK1 FOREIGN KEY (D_SEQ) REFERENCES DEVICE(D_SEQ)
);
CREATE SEQUENCE DEVICEINFO_SEQ;
